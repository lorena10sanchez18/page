public int[] insert(int[] elemento) {
int auxiliar;
for (int M = 1; M < elemento.length; M++) {
auxiliar = elemento[M];
for (int S = M - 1; S >= 0 && elemento[S] > auxiliar; S--) {
elemento[S + 1] = elemento[S];
elemento[S] = auxiliar;
}
}
return elemento;
}
