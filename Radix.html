// The main function to that sorts arr[] of size n using
// Radix Sort
static void radixsort(int arr[], int n)
{
// Buscar el número máximo para saber el número de dígitos
int m = getMax(arr, n);
// Hacer el conteo de clasificación para cada dígito. Tenga en cuenta que en su lugar
// de pasar número de dígito, exp se pasa. Exp es 10 ^ i
// donde i es el número de dígito actual
for (int exp = 1; m/exp > 0; exp *= 10)
countSort(arr, n, exp);
}
// Una función para hacer contando tipo de array [] según
// el dígito representado por exp.
static void countSort(int array[], int n, int exp)
{
int salida[] = new int[n]; // salida array
int i;
int count[] = new int[10];
Arrays.fill(count,0);
// Almacenar el número de ocurrencias en el conteo []
for (i = 0; i < n; i++)
count[ (array[i]/exp)%10 ]++;
// Cambiar el recuento [i] para que el recuento [i] contenga ahora
// posición real de este dígito en salida []
for (i = 1; i < 10; i++)
count[i] += count[i - 1];
// Construir el array de salida
for (i = n - 1; i >= 0; i--)
{
salida[count[ (array[i]/exp)%10 ] - 1] = array[i];
count[ (array[i]/exp)%10 ]--;
}
// Copie el array de salida a array [], de modo que array [] ahora
// contiene números clasificados según el dígito actual
for (i = 0; i < n; i++)
array[i] = salida[i];
}
static int getMax(int array[], int n)
{
int mx = array[0];
for (int i = 1; i < n; i++)
if (array[i] > mx)
mx = array[i];
return mx;
}
