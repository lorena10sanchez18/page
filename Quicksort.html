Metodos de Ordenamiento

Este es el algoritmo de metodo de ordenamiento Busqueda Rapida

public class QuickSort {
private int array[];
private int tamaño;
public static void main(String a[]) {
QuickSort clasificador = new QuickSort();
int[] entrada = {24, 2, 45, 20, 46, 75, 2, 56, 99, 53, 12};
clasificador.ordenar(entrada);
for (int i : entrada) {
System.out.print(i);
System.out.print(" ");
}
private void quickSort(int indiceInferior, int indiceSuperior) {
int i = indiceInferior;
int j = indiceSuperior;
// Calcular el número de pivote, estoy tomando pivote como número de índice medio
int pivote = array[indiceInferior + (indiceSuperior - indiceInferior) / 2];
// Divide en dos arrays
while (i <= j) {
/**
* En cada iteración, identificaremos un número del lado izquierdo que
* Es mayor que el valor de pivote, y también vamos a identificar un número
* Desde el lado derecho que es menor que el valor de pivote. Una vez que la búsqueda
* Se hace, entonces intercambiamos ambos números.
*/
while (array[i] < pivote) {
i++;
}
while (array[j] > pivote) {
j--;
}
if (i <= j) {
numerosDeIntercambio(i, j);
//Mover el índice a la siguiente posición en ambos lados
i++;
j--;
}
}
// Llamada quickSort () método recursivamente
if (indiceInferior < j)
quickSort(indiceInferior, j);
if (i < indiceSuperior)
quickSort(i, indiceSuperior);
}
private void numerosDeIntercambio(int i, int j) {
int temp = array[i];
array[i] = array[j];
array[j] = temp;
}
public void ordenar(int[] arrayEntrada) {
if (arrayEntrada == null || arrayEntrada.length == 0) {
return;
}
this.array = arrayEntrada;
tamaño = arrayEntrada.length;
quickSort(0, tamaño - 1);
}
}
